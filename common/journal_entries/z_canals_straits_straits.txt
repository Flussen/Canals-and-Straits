je_cs_bosphorus_control = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"
	group = je_group_historical_content
	weight = 100
	should_be_pinned_by_default = yes
	scripted_button = cs_button_bosphorus_open
	scripted_button = cs_button_bosphorus_tolls
	scripted_button = cs_button_bosphorus_blockade
	is_shown_when_inactive = {
		OR = {
			has_interest_marker_in_region = region_balkans
			has_interest_marker_in_region = region_anatolia
			has_interest_marker_in_region = region_russia
		}
	}
	possible = {
		NOT = { is_subject = yes }
		OR = {
			country_or_subject_owns_entire_state_region = STATE_EASTERN_THRACE
			any_scope_state = { state_region = s:STATE_EASTERN_THRACE }
		}
	}
	complete = { always = no }
}

je_cs_kiel_survey = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"
	group = je_group_technology
	on_monthly_pulse = {
		effect = {
			change_variable = { name = survey_kiel_taken add = 1 }
		}
	}
	complete = { scope:journal_entry = { is_goal_complete = yes } }
	on_complete = {
		remove_modifier = modifier_surveying_kiel
		remove_variable = survey_kiel_taken
		set_variable = kiel_survey_complete
	}
	current_value = { value = root.var:survey_kiel_taken }
	goal_add_value = { value = 12 }
	invalid = {
		OR = {
			NOT = { has_variable = survey_kiel_taken }
			s:STATE_SCHLESWIG_HOLSTEIN = {
				any_scope_state = {
					NOT = { owner = ROOT }
					has_building = building_kiel_canal
				}
			}
		}
	}
	on_invalid = {
		remove_variable = survey_kiel_taken
		remove_modifier = modifier_surveying_kiel
	}
	progressbar = yes
	weight = 10
	should_be_pinned_by_default = yes
}
