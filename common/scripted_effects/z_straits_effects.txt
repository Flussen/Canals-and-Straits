cs_monthly_reapply_maritime_policies = {
	save_scope_as = cs_controller
	if = {
		limit = {
			has_modifier = cs_gibraltar_control
		}
		remove_modifier = cs_gibraltar_control
	}
	if = {
		limit = {
			has_modifier = cs_bosphorus_control
		}
		remove_modifier = cs_bosphorus_control
	}
	if = {
		limit = {
			has_modifier = cs_malacca_control
		}
		remove_modifier = cs_malacca_control
	}
	if = {
		limit = {
			has_modifier = cs_hormuz_control
		}
		remove_modifier = cs_hormuz_control
	}
	if = {
		limit = {
			has_modifier = cs_hormuz_regional_primacy
		}
		remove_modifier = cs_hormuz_regional_primacy
	}
	if = {
		limit = {
			OR = {
				any_scope_state = { state_region = s:STATE_LOWER_ANDALUSIA }
				any_scope_state = { state_region = s:STATE_UPPER_ANDALUSIA }
				any_subject_or_below = { any_scope_state = { state_region = s:STATE_LOWER_ANDALUSIA } }
				any_subject_or_below = { any_scope_state = { state_region = s:STATE_UPPER_ANDALUSIA } }
				country_or_subject_owns_entire_state_region = STATE_AL_RIF
			}
		}
		add_modifier = {
			name = cs_gibraltar_control
			days = 45
		}
		if = { limit = { has_modifier = cs_gibraltar_policy_open } remove_modifier = cs_gibraltar_policy_open }
		if = { limit = { has_modifier = cs_gibraltar_policy_tolls } remove_modifier = cs_gibraltar_policy_tolls }
		if = { limit = { has_modifier = cs_gibraltar_policy_blockade } remove_modifier = cs_gibraltar_policy_blockade }
		if = {
			limit = { has_variable = cs_gibraltar_policy_open_var }
			add_modifier = { name = cs_gibraltar_policy_open days = 45 }
		}
		else_if = {
			limit = { has_variable = cs_gibraltar_policy_blockade_var }
			add_modifier = { name = cs_gibraltar_policy_blockade days = 45 }
		}
		else = {
			add_modifier = {
				name = cs_gibraltar_policy_tolls
				days = 45
			}
		}
	}
	if = {
		limit = {
			country_or_subject_owns_entire_state_region = STATE_EASTERN_THRACE
		}
		add_modifier = {
			name = cs_bosphorus_control
			days = 45
		}
		if = { limit = { has_modifier = cs_bosphorus_policy_open } remove_modifier = cs_bosphorus_policy_open }
		if = { limit = { has_modifier = cs_bosphorus_policy_tolls } remove_modifier = cs_bosphorus_policy_tolls }
		if = { limit = { has_modifier = cs_bosphorus_policy_blockade } remove_modifier = cs_bosphorus_policy_blockade }
		if = {
			limit = { has_variable = cs_bosphorus_policy_open_var }
			add_modifier = { name = cs_bosphorus_policy_open days = 45 }
		}
		else_if = {
			limit = { has_variable = cs_bosphorus_policy_blockade_var }
			add_modifier = { name = cs_bosphorus_policy_blockade days = 45 }
		}
		else = {
			add_modifier = {
				name = cs_bosphorus_policy_tolls
				days = 45
			}
		}
	}
	if = {
		limit = {
			country_or_subject_owns_entire_state_region = STATE_MALAYA
		}
		add_modifier = {
			name = cs_malacca_control
			days = 45
		}
		if = { limit = { has_modifier = cs_malacca_policy_open } remove_modifier = cs_malacca_policy_open }
		if = { limit = { has_modifier = cs_malacca_policy_tolls } remove_modifier = cs_malacca_policy_tolls }
		if = { limit = { has_modifier = cs_malacca_policy_blockade } remove_modifier = cs_malacca_policy_blockade }
		if = {
			limit = { has_variable = cs_malacca_policy_open_var }
			add_modifier = { name = cs_malacca_policy_open days = 45 }
		}
		else_if = {
			limit = { has_variable = cs_malacca_policy_blockade_var }
			add_modifier = { name = cs_malacca_policy_blockade days = 45 }
		}
		else = {
			add_modifier = {
				name = cs_malacca_policy_tolls
				days = 45
			}
		}
	}
	if = {
		limit = {
			country_or_subject_owns_entire_state_region = STATE_LARISTAN
		}
		add_modifier = {
			name = cs_hormuz_control
			days = 45
		}
		if = { limit = { has_modifier = cs_hormuz_policy_open } remove_modifier = cs_hormuz_policy_open }
		if = { limit = { has_modifier = cs_hormuz_policy_tolls } remove_modifier = cs_hormuz_policy_tolls }
		if = { limit = { has_modifier = cs_hormuz_policy_blockade } remove_modifier = cs_hormuz_policy_blockade }
		if = {
			limit = { has_variable = cs_hormuz_policy_open_var }
			add_modifier = { name = cs_hormuz_policy_open days = 45 }
		}
		else_if = {
			limit = { has_variable = cs_hormuz_policy_blockade_var }
			add_modifier = { name = cs_hormuz_policy_blockade days = 45 }
		}
		else = {
			add_modifier = {
				name = cs_hormuz_policy_tolls
				days = 45
			}
		}
		if = {
			limit = {
				this = c:PER
			}
			add_modifier = {
				name = cs_hormuz_regional_primacy
				days = 45
			}
		}
	}
	cs_apply_suez_effects = yes
	cs_apply_panama_effects = yes
	cs_apply_kiel_effects = yes
	every_country = {
		limit = {
			NOT = {
				THIS = scope:cs_controller
			}
			any_scope_state = {
				any_scope_building = {
					is_building_type = building_port
				}
			}
		}
		if = {
			limit = {
				scope:cs_controller = {
					has_modifier = cs_gibraltar_policy_blockade
				}
				scope:cs_controller = {
					OR = {
						any_scope_state = { state_region = s:STATE_LOWER_ANDALUSIA }
						any_scope_state = { state_region = s:STATE_UPPER_ANDALUSIA }
						any_subject_or_below = { any_scope_state = { state_region = s:STATE_LOWER_ANDALUSIA } }
						any_subject_or_below = { any_scope_state = { state_region = s:STATE_UPPER_ANDALUSIA } }
						country_or_subject_owns_entire_state_region = STATE_AL_RIF
					}
				}
				has_war_with = scope:cs_controller
			}
			add_modifier = {
				name = cs_strait_blocked
				days = 45
			}
		}
		if = {
			limit = {
				scope:cs_controller = {
					has_modifier = cs_bosphorus_policy_blockade
				}
				scope:cs_controller = {
					country_or_subject_owns_entire_state_region = STATE_EASTERN_THRACE
				}
				has_war_with = scope:cs_controller
			}
			add_modifier = {
				name = cs_strait_blocked
				days = 45
			}
		}
		if = {
			limit = {
				scope:cs_controller = {
					has_modifier = cs_malacca_policy_blockade
				}
				scope:cs_controller = {
					country_or_subject_owns_entire_state_region = STATE_MALAYA
				}
				has_war_with = scope:cs_controller
			}
			add_modifier = {
				name = cs_strait_blocked
				days = 45
			}
		}
		if = {
			limit = {
				scope:cs_controller = {
					has_modifier = cs_hormuz_policy_blockade
				}
				scope:cs_controller = {
					country_or_subject_owns_entire_state_region = STATE_LARISTAN
				}
				has_war_with = scope:cs_controller
			}
			add_modifier = {
				name = cs_strait_blocked
				days = 45
			}
		}
	}
}
