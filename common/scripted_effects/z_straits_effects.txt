cs_monthly_reapply_maritime_policies = {
	save_scope_as = cs_controller
	
	# Gibraltar
	if = {
		limit = { has_modifier = cs_gibraltar_control }
		remove_modifier = cs_gibraltar_control
	}
	if = {
		limit = { has_modifier = cs_gibraltar_control_both }
		remove_modifier = cs_gibraltar_control_both
	}
	if = {
		limit = {
			OR = {
				any_scope_state = { state_region = s:STATE_LOWER_ANDALUSIA }
				any_scope_state = { state_region = s:STATE_UPPER_ANDALUSIA }
				any_subject_or_below = { any_scope_state = { state_region = s:STATE_LOWER_ANDALUSIA } }
				any_subject_or_below = { any_scope_state = { state_region = s:STATE_UPPER_ANDALUSIA } }
				country_or_subject_owns_entire_state_region = STATE_AL_RIF
			}
		}
		if = {
			limit = {
				OR = {
					country_or_subject_owns_entire_state_region = STATE_AL_RIF
					any_subject_or_below = { country_or_subject_owns_entire_state_region = STATE_AL_RIF }
				}
				OR = {
					any_scope_state = { state_region = s:STATE_LOWER_ANDALUSIA }
					any_scope_state = { state_region = s:STATE_UPPER_ANDALUSIA }
					any_subject_or_below = { any_scope_state = { state_region = s:STATE_LOWER_ANDALUSIA } }
					any_subject_or_below = { any_scope_state = { state_region = s:STATE_UPPER_ANDALUSIA } }
				}
			}
			add_modifier = {
				name = cs_gibraltar_control_both
				days = 45
			}
		}
		else = {
			add_modifier = {
				name = cs_gibraltar_control
				days = 45
			}
		}
	}
	
	# Bosphorus
	if = {
		limit = { has_modifier = cs_bosphorus_control }
		remove_modifier = cs_bosphorus_control
	}
	if = {
		limit = { has_modifier = cs_bosphorus_control_both }
		remove_modifier = cs_bosphorus_control_both
	}
	if = {
		limit = {
			OR = {
				country_or_subject_owns_entire_state_region = STATE_EASTERN_THRACE
				any_scope_state = { state_region = s:STATE_EASTERN_THRACE }
			}
		}
		if = {
			limit = {
				OR = {
					country_or_subject_owns_entire_state_region = STATE_EASTERN_THRACE
					any_scope_state = { state_region = s:STATE_EASTERN_THRACE }
				}
				OR = {
					country_or_subject_owns_entire_state_region = STATE_WESTERN_THRACE
					any_scope_state = { state_region = s:STATE_WESTERN_THRACE }
					any_subject_or_below = { country_or_subject_owns_entire_state_region = STATE_WESTERN_THRACE }
				}
			}
			add_modifier = {
				name = cs_bosphorus_control_both
				days = 45
			}
		}
		else = {
			add_modifier = {
				name = cs_bosphorus_control
				days = 45
			}
		}
	}
	
	# Malacca
	if = {
		limit = { has_modifier = cs_malacca_control }
		remove_modifier = cs_malacca_control
	}
	if = {
		limit = { has_modifier = cs_malacca_control_both }
		remove_modifier = cs_malacca_control_both
	}
	if = {
		limit = {
			country_or_subject_owns_entire_state_region = STATE_MALAYA
		}
		if = {
			limit = {
				country_or_subject_owns_entire_state_region = STATE_MALAYA
				OR = {
					country_or_subject_owns_entire_state_region = STATE_SUMATRA
					any_scope_state = { state_region = s:STATE_SUMATRA }
					any_subject_or_below = { country_or_subject_owns_entire_state_region = STATE_SUMATRA }
				}
			}
			add_modifier = {
				name = cs_malacca_control_both
				days = 45
			}
		}
		else = {
			add_modifier = {
				name = cs_malacca_control
				days = 45
			}
		}
	}
	
	# Hormuz
	if = {
		limit = { has_modifier = cs_hormuz_control }
		remove_modifier = cs_hormuz_control
	}
	if = {
		limit = { has_modifier = cs_hormuz_control_both }
		remove_modifier = cs_hormuz_control_both
	}
	if = {
		limit = { has_modifier = cs_hormuz_regional_primacy }
		remove_modifier = cs_hormuz_regional_primacy
	}
	if = {
		limit = {
			country_or_subject_owns_entire_state_region = STATE_LARISTAN
		}
		if = {
			limit = {
				country_or_subject_owns_entire_state_region = STATE_LARISTAN
				OR = {
					country_or_subject_owns_entire_state_region = STATE_ABU_DHABI
					any_scope_state = { state_region = s:STATE_ABU_DHABI }
					any_subject_or_below = { country_or_subject_owns_entire_state_region = STATE_ABU_DHABI }
				}
			}
			add_modifier = {
				name = cs_hormuz_control_both
				days = 45
			}
		}
		else = {
			add_modifier = {
				name = cs_hormuz_control
				days = 45
			}
		}
		if = {
			limit = { this = c:PER }
			add_modifier = {
				name = cs_hormuz_regional_primacy
				days = 45
			}
		}
	}
	
	cs_apply_suez_effects = yes
	cs_apply_panama_effects = yes
	cs_apply_kiel_effects = yes
}
