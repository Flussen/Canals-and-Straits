cs_monthly_country_pulse = {
	effect = {
		every_country = {
			if = { limit = { has_modifier = cs_strait_blocked } remove_modifier = cs_strait_blocked }
			if = { limit = { has_modifier = cs_panama_transit_toll } remove_modifier = cs_panama_transit_toll }
			if = { limit = { has_modifier = cs_panama_transit_blocked } remove_modifier = cs_panama_transit_blocked }
			if = { limit = { has_modifier = cs_suez_transit_toll } remove_modifier = cs_suez_transit_toll }
			if = { limit = { has_modifier = cs_suez_transit_blocked } remove_modifier = cs_suez_transit_blocked }
			if = { limit = { has_modifier = cs_kiel_transit_toll } remove_modifier = cs_kiel_transit_toll }
			if = { limit = { has_modifier = cs_kiel_transit_blocked } remove_modifier = cs_kiel_transit_blocked }
		}
		every_country = {
			cs_monthly_reapply_maritime_policies = yes
		}
	}
}

on_monthly_pulse = {
	on_actions = {
		cs_monthly_country_pulse
		cs_canal_economic_pulse
	}
}
